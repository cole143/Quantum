def add_financial_disclaimer(output_text, output_type):
    """

FDIC insurance information is current as of November 2025 and subject to change.
Standard coverage: $250,000 per depositor, per bank, per ownership category.
Verify coverage at www.fdic.gov/EDIE or with your financial institution.

Consult your financial institution or banking attorney for specific guidance.
        """,

⚠️ DISCLAIMER:
Information provided is for educational purposes only and not professional financial, 
tax, or legal advice. Quantum assumes no liability for decisions made based on this 
information. Consult qualified professionals for your specific situation.
        """
    }
    
    disclaimer = disclaimers.get(output_type, disclaimers['general'])
    return output_text + disclaimer


def validate_financial_output(output_text):
    """
    Checks if output contains required disclaimers and compliance language
    
    Returns warnings if critical elements are missing
    """
    
    warnings = []
    
    # Check for disclaimer presence
    if "⚠️" not in output_text and "disclaimer" not in output_text.lower():
        warnings.append("WARNING: No disclaimer detected in financial output")
    
    # Check for prohibited language
    prohibited_phrases = [
        "i recommend you",
        "you should invest",
        "as your financial advisor",
        "guaranteed returns",
        "no risk",
        "trust me"
    ]
    
    for phrase in prohibited_phrases:
        if phrase in output_text.lower():
            warnings.append(f"WARNING: Prohibited phrase detected: '{phrase}'")
    
    # Check for tax calculation without disclaimer
    if "tax" in output_text.lower() and "bracket" in output_text.lower():
        if "estimate only" not in output_text.lower():
            warnings.append("WARNING: Tax calculation missing 'estimate only' language")
    
    # Check for FDIC mention without disclaimer
    if "fdic" in output_text.lower():
        if "fdic.gov" not in output_text.lower():
            warnings.append("WARNING: FDIC mentioned without reference to fdic.gov")
    
    return warnings
